572 void CCodeBlock::CleanupControlFlow(void)
573 {
574  list<CTacInstr*>::iterator it = _ops.begin();
575 
576  // 1. pass: delete all branches (absolute/conditional) that jump to the
577  // immediately next instruction. Deleting branch instruction will
578  // automatically decrease the reference count of the target label.
579  while (it != _ops.end()) {
580  CTacInstr *instr = *it++;
581 
582  if (instr->IsBranch()) {
583  CTacLabel *lbl = dynamic_cast<CTacLabel*>(instr->GetDest());
584  CTacInstr *next = (it == _ops.end() ? NULL : *it);
585 
586  if ((lbl != NULL) && (lbl == next)) {
587  delete instr;
588  it = _ops.erase(--it);
589  }
590  }
591  }
592 
593  // 2. pass: remove all labels with reference count 0
594  it = _ops.begin();
595  while (it != _ops.end()) {
596  CTacInstr *instr = *it++;
597 
598  CTacLabel *lbl = dynamic_cast<CTacLabel*>(instr);
599 
600  if ((lbl != NULL) && (lbl->GetRefCnt() == 0)) {
601  delete lbl;
602  it = _ops.erase(--it);
603  }
604  }
605 
606  // 3. renumber instructions (we shouldn't do that really, but it's prettier)
607  _inst_id = 0;
608  it = _ops.begin();
609  while (it != _ops.end()) (*it++)->SetId(_inst_id++);
610 }
