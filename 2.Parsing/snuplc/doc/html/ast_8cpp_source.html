1166  if(boolConst->GetValue() == 1) // true
1167  cb->AddInstr(new CTacInstr(opGoto, l_true));
1168  else // false
1169  cb->AddInstr(new CTacInstr(opGoto, l_false));
1170  }
1171  //else if(dynamic_cast<CAstDesignator*>(right)) { // right term is id.
1172  else if(tRight) { // if right term is id or func call, then tRight is not null
1173  cb->AddInstr(new CTacInstr(opEqual, l_true, tRight, new CTacConst(1)));
1174  cb->AddInstr(new CTacInstr(opGoto, l_false));
1175  }
1176  break;
1177  case opAnd:
1178  tLeft = left->ToTac(cb, test_next, l_false);
1179  // left term is boolean constant.
1180  if(CAstConstant* boolConst = dynamic_cast<CAstConstant*>(left)) {
1181  if(boolConst->GetValue() == 1) // true
1182  cb->AddInstr(new CTacInstr(opGoto, test_next));
1183  else // false
1184  cb->AddInstr(new CTacInstr(opGoto, l_false));
1185  }
1186  //else if(dynamic_cast<CAstDesignator*>(left)) { // left term is id.
1187  else if(tLeft) { // if left term is id or func call, then tLeft is not null
1188  cb->AddInstr(new CTacInstr(opEqual, test_next, tLeft, new CTacConst(1)));
1189  cb->AddInstr(new CTacInstr(opGoto, l_false));
1190  }
1191  cb->AddInstr(test_next);
1192  tRight = right->ToTac(cb, l_true, l_false);
1193  // right term is boolean constant.
1194  if(CAstConstant* boolConst = dynamic_cast<CAstConstant*>(right)) {
1195  if(boolConst->GetValue() == 1) // true
1196  cb->AddInstr(new CTacInstr(opGoto, l_true));
1197  else // false
1198  cb->AddInstr(new CTacInstr(opGoto, l_false));
1199  }
1200  //else if(dynamic_cast<CAstDesignator*>(right)) { // right term is id.
1201  else if(tRight) { // if right term is id or func call, then tRight is not null
1202  cb->AddInstr(new CTacInstr(opEqual, l_true, tRight, new CTacConst(1)));
1203  cb->AddInstr(new CTacInstr(opGoto, l_false));
1204  }
1205  break;
1206  // rel Ops
1207  case opEqual:
1208  case opNotEqual:
1209  case opLessThan:
1210  case opLessEqual:
1211  case opBiggerThan:
1212  case opBiggerEqual:

1214  tRight = right->ToTac(cb, l_true, l_false);
1215  cb->AddInstr(new CTacInstr(oper, l_true, tLeft, tRight));
1216  cb->AddInstr(new CTacInstr(opGoto, l_false));
1217  break;
1218  }
1219  if(!ltrue) { // assign temp as true
1220  cb->AddInstr(l_true);
1221  cb->AddInstr(new CTacInstr(opAssign, ret, new CTacConst(1)));
1222  cb->AddInstr(new CTacInstr(opGoto, l_next));

1226  cb->AddInstr(new CTacInstr(opAssign, ret, new CTacConst(0)));
1227  cb->AddInstr(new CTacInstr(opGoto, l_next));
1228  }
1229  if(!ltrue || !lfalse)
1230  cb->AddInstr(l_next);
1231  }
1232  else { // related with integer
1233  // no need for label of true & false
1234  tLeft = left->ToTac(cb);
1235  tRight = right->ToTac(cb);
1236 
1237  ret = cb->CreateTemp(left->GetType()); // left & right same type.
1238  cb->AddInstr(new CTacInstr(oper, ret, tLeft, tRight));
1239  }
1240 
1241  return ret;
1242 }
