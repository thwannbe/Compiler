125 {
126  //
127  // module = "module" ident ";" varDeclaration { subroutineDecl }
128  // "begin" statSequence "end" ident ".".
129  //
130  // FIRST(module) = { tModule }
131  // FOLLOW(module) = { $ }
132  //
133  CToken mod;
134  Consume(tModule, &mod);
135

138  CAstModule *m = new CAstModule(mod, modName.GetValue());
139  InitSymbolTable(m->GetSymbolTable());
140 
141  // As module class is constructed, it initialize symbol table too.
142  Consume(tSemicolon);
143 
144  // global variable is added at symbol table
145  varDeclaration(m);
146 
147  EToken tt = _scanner->Peek().GetType();
148  if(tt != tBegin) { // means subroutineDecl exists
149  do {
150  switch (tt) {
151  case tProcedure:
152  case tFunction:
153  subroutineDecl(m);
154  break;
155 
156  default:
157  SetError(_scanner->Peek(), "subroutineDecl expected.");
158  break;
159  }
160 
161  tt = _scanner->Peek().GetType();
162  if(tt == tBegin) break;
163 
164  } while (!_abort);
165  }
166  Consume(tBegin);
167 
168  CAstStatement *statseq = NULL;
169  statseq = statSequence(m);
170 
171  Consume(tEnd);

173  SetError(_scanner->Peek(), "module identifier does not match.");

182 void CParser::varDeclaration(CAstScope *s)
183 {
184  //
185  // varDeclaration = [ "var" { ident { "," ident } ":" type ";" } ].
186  //
187  // FIRST(varDeclaration) = { tVarDecl, e }
188  // FOLLOW(varDeclaration) = { tProcedure, tFunction, tBegin }
189  //
190  EToken tt = _scanner->Peek().GetType();
191  if(tt == tVarDecl) {
192  Consume(tVarDecl);
193  do {
194  CToken t;
195  vector<CSymbol*> _gl_var;
196  tt = _scanner->Peek().GetType();
197  if(tt == tProcedure || tt == tFunction || tt == tBegin) break;
198 
199  Consume(tIdent, &t);
200  if(s->GetSymbolTable()->FindSymbol(t.GetValue(), sLocal))
201  SetError(_scanner->Peek(), "Symbol [" + t.GetValue() + "] is already declared.");
202  _gl_var.push_back(s->CreateVar(t.GetValue(), CTypeManager::Get()->GetNull()));
203

206  if (tt != tComma) break;
207

209  Consume(tIdent, &t);
210  if(s->GetSymbolTable()->FindSymbol(t.GetValue(), sLocal))
211  SetError(_scanner->Peek(), "Symbol [" + t.GetValue() + "] is already declared.");
212  _gl_var.push_back(s->CreateVar(t.GetValue(), CTypeManager::Get()->GetNull()));
213  } while (!_abort);
214 
215  Consume(tColon);
216 
217  tt = _scanner->Peek().GetType();
218  switch(tt) {
219  case tInteger:
220  Consume(tInteger);
221  for(int i = 0; i < _gl_var.size(); i++)
222  _gl_var[i]->SetDataType(CTypeManager::Get()->GetInt());
223  break;
224

226  Consume(tBoolean);
227  for(int i = 0; i < _gl_var.size(); i++)
228  _gl_var[i]->SetDataType(CTypeManager::Get()->GetBool());
229  break;

233  break;
234  }
235  Consume(tSemicolon);

404 CAstStatement* CParser::statSequence(CAstScope *s)
405 {
406  //
407  // statSequence = [ statement { ";" statement } ].
408  // statement = assignment | subroutineCall | ifStatement |
409  // whileStatement | returnStatement.
410  //
411  // FIRST(statSequence) = { tIdent, tIf, tWhile, tReturn, e }
412  // FOLLOW(statSequence) = { tEnd, tElse }
413  //
414  CAstStatement *head = NULL;
415 
416  EToken tt = _scanner->Peek().GetType();
417  if (tt != tEnd && tt != tElse) {
418  CAstStatement *tail = NULL;
419 
420  do {
421  CToken t;
422  tt = _scanner->Peek().GetType();
423  CAstStatement *st = NULL;
424  CAstDesignator *id = NULL;
425 
426  switch (tt) {
427  case tIdent: // assignment or subroutineCall
428  id = ident(s);
429  assert(id != NULL);
430  tt = _scanner->Peek().GetType();
431  if (tt == tAssign) {
432  if(id->GetSymbol()->GetSymbolType() == stProcedure)
433  SetError(_scanner->Peek(), "subroutine symbol can't be assigned.");
434  st = assignment(s, id);
435  } else if (tt == tLBrak) {
436  st = subroutineCall(s, id);
437  } else {
438  SetError(_scanner->Peek(), "assignment or subroutinecall expected.");
439  }
440  break;
441 
442  case tIf:
443  st = ifStatement(s);
444  break;
445 
446  case tWhile:
447  st = whileStatement(s);
448  break;
449 
450  case tReturn:
451  st = returnStatement(s);
452  break;
453 
454  default:
455  SetError(_scanner->Peek(), "statement expected.");
456  break;
457  }
458  assert(st != NULL);
459  if (head == NULL) head = st;
460  else tail->SetNext(st);
461  tail = st;
462 
463  tt = _scanner->Peek().GetType();
464  if (tt == tEnd || tt == tElse) break;
465 
466  Consume(tSemicolon);
467  } while (!_abort);
468  }

617 {

619  // expression = simpleexpr [ relOp simpleexpr ].
620  //
621  // FIRST(expression) = { tIdent, tNumber, tBoolConst, tLBrak, tNot, tPlusMinus }

737 CAstExpression* CParser::factor(CAstScope *s)
738 {
739  //
740  // factor = ident | number | boolean | "(" expression ")" |
741  // subroutineCall | "!" factor.
742  //
743  // FIRST(factor) = { tIdent, tNumber, tBoolConst, tLBrak, tNot }
744  // FOLLOW(factor) = { tMulDiv, tAnd, tPlusMinus, tOr, tRelOp, tComma, tRBrak,
745  // tSemicolon, tEnd, tElse }
746  //
747  CToken t;
748  CAstExpression *n = NULL;
749 
750  CAstExpression *e = NULL; // for factor ::= "!" factor
751  CAstStatCall *call = NULL; // for subroutineCall
752 
753  EToken tt = _scanner->Peek().GetType();
754 
755  CAstDesignator *id = NULL;
756  switch (tt) {
757  // factor ::= ident | subroutineCall
758  case tIdent:
759  id = ident(s);
760  tt = _scanner->Peek().GetType();
761  if(tt == tLBrak) {
762  call = subroutineCall(s, id);
763  n = call->GetCall();
764  } else {
765  n = id;
766  }
767  break;
768 
769  // factor ::= number
770  case tNumber:
771  n = number();
772  break;
773 
774  // factor ::= boolean
775  case tBoolConst:
776  n = boolean();
777  break;
778 
779  // factor ::= "(" expression ")"
780  case tLBrak:
781  Consume(tLBrak);
782  n = expression(s);
783  Consume(tRBrak);
784  break;
785
