175 void CBackendx86::EmitScope(CScope *scope)
176 {
177  assert(scope != NULL);
178 
179  string label;
180 
181  if (scope->GetParent() == NULL) label = "main";
182  else label = scope->GetName();
183 
184  // label
185  _out << _ind << "# scope " << scope->GetName() << endl
186  << label << ":" << endl;
187 
188  SetScope(scope);
189  // param_offs = 8, local_offs = -12 (details on ComputeStackOffsets's comment)
190  size_t locSize = ComputeStackOffsets(scope->GetSymbolTable(), 8, -12);
191 
192  // Subroutine Prologue
193  // 1. save ebp by pushing it onto the stack
194  // 2. set ebp to esp
195  // 3. save callee-saved registers
196  // 4. generated space on the stack for locals and spilled variables by adjusting
197  // the stack pointer
198 
199  EmitInstruction("pushl", "%ebp");
200  EmitInstruction("movl" , "%esp, %ebp");
201  EmitInstruction("pushl", "%ebx");
202  EmitInstruction("pushl", "%esi");
203  EmitInstruction("pushl", "%edi");
204  EmitInstruction("subl" , Imm(locSize) + string(", %esp"),
205  "make room for locals");
206  _out << endl;
207 
208  EmitCodeBlock(scope->GetCodeBlock(), scope->GetSymbolTable());
209  _out << endl;
210 
211  // Subroutine Epilogue
212  // 1. remove space on stack for locals and spilled variables by setting
213  // the stack pointer immediately below the callee-saved registers
214  // 2. restore callee-saved registers
215  // 3. restore ebp
216  // 4. issue the ret instruction
217 
218  label = Label("exit");
219  _out << label << ":" << endl;
220  EmitInstruction("addl", Imm(locSize) + string(", %esp"),
221  "remove locals");
222  EmitInstruction("popl", "%edi");
223  EmitInstruction("popl", "%esi");
224  EmitInstruction("popl", "%ebx");
225  EmitInstruction("popl", "%ebp");
226  EmitInstruction("ret");

228  _out << endl;

285 void CBackendx86::EmitInstruction(CTacInstr *i, CSymtab *symtab)
286 {
287  assert(i != NULL);
288  assert(symtab != NULL);
289 
290  ostringstream cmt;
291  cmt << i;
292 
293  EOperation op = i->GetOperation();
294  CTac *src1 = i->GetSrc(1); // can be null
295  CTac *src2 = i->GetSrc(2); // can be null
296  CTac *dst = i->GetDest(); // can be null
297 
298  // for opCall
299  CTacName *callee = NULL;
300  const CSymProc *calleeSym = NULL;
301 
302  switch (op) {
303  // binary operators
304  case opAdd:
305  case opSub:
306  case opAnd:
307  case opOr:
308  EmitInstruction("movl", Operand(src1) + string(", %eax"),
309  cmt.str());
310  EmitInstruction(Operation(op), Operand(src2) + string(", %eax"));
311  EmitInstruction("movl", string("%eax, ") + Operand(dst));
312  break;
313 
314  case opMul:
315  case opDiv:
316  // Multiply and Divide uses fixed registers %eax, %edx in X86 arch
317  EmitInstruction("movl", Operand(src1) + string(", %eax"),
318  cmt.str());
319  // %edx register should be empty before division
320  if (op == opDiv)
321  EmitInstruction("cdq"); // sign extension %eax => [%edx:%eax]
322  // Multiply and Divide should move second operand to %ebx, and operate
323  EmitInstruction("movl", Operand(src2) + string(", %ebx"));
324  EmitInstruction(Operation(op), "%ebx");
325  EmitInstruction("movl", string("%eax, ") + Operand(dst));
326  break;
327 
328  // unary operators
329  case opNot: // opNot is not used at all
330  case opNeg:
331  EmitInstruction("movl", Operand(src1) + string(", %eax"),
332  cmt.str());
333  EmitInstruction(Operation(op), "%eax");
334  EmitInstruction("movl", string("%eax, ") + Operand(dst));
335  break;
336 
337  case opPos:
338  // memory operations
339  case opAssign:
340  // Pos unary op just do copy like Assgin op
341  if(dynamic_cast<CTacName*>(dst)->GetSymbol()->GetDataType() ==
342  CTypeManager::Get()->GetBool()) {
343  EmitInstruction("movb", Operand(src1) + string(", %al"), cmt.str());
344  EmitInstruction("movb", string("%al, ") + Operand(dst));
345  }
346  else {// int
347  EmitInstruction("movl", Operand(src1) + string(", %eax"), cmt.str());
348  EmitInstruction("movl", string("%eax, ") + Operand(dst));
349  }
350  break;
351 
352  // unconditional branching
353  case opGoto:
354  EmitInstruction(Operation(op), Label(dynamic_cast<CTacLabel*>(dst)),
355  cmt.str());
356  break;
357 
358  // conditional branching
359  case opEqual:
360  case opNotEqual:
361  case opLessThan:
362  case opLessEqual:
363  case opBiggerThan:
364  case opBiggerEqual:
365  EmitInstruction("movl", Operand(src1) + string(", %eax"),
366  cmt.str());
367  EmitInstruction("cmpl", Operand(src2) + string(", %eax"));
368  EmitInstruction(Operation(op), Label(dynamic_cast<CTacLabel*>(dst)));
369  break;
370 
371  // function call-related operations
372  case opParam:
373  EmitInstruction("movl", Operand(src1) + string(", %eax"),
374  cmt.str());
375  EmitInstruction("pushl", "%eax");
376  break;
377 
378  case opCall:
379  callee = dynamic_cast<CTacName*>(src1);
380  calleeSym = dynamic_cast<const CSymProc*>(callee->GetSymbol());
381  EmitInstruction(Operation(op), Operand(src1), cmt.str());
382  // each param has 4 byte (int type)
383  if(calleeSym->GetNParams()) // not 0 param
384  EmitInstruction("addl", Imm(calleeSym->GetNParams() * 4) + string(", %esp"));
385  if(dst)
386  EmitInstruction("movl", string("%eax, ") + Operand(dst));
387  break;
388 
389  // return
390  case opReturn:
391  if(src1)
392  EmitInstruction("movl", Operand(src1) + string(", %eax"), cmt.str());
393  EmitInstruction("jmp", Label("exit"));
394  break;
395 
396  // special
397  case opLabel:
398  _out << Label(dynamic_cast<CTacLabel*>(i)) << ":" << endl;
399  break;
400 
401  case opNop:
402  EmitInstruction("nop", "", cmt.str());
403  break;
404 
405  default:
406  EmitInstruction("# ???", "not implemented", cmt.str());
407  }

444

537 size_t CBackendx86::ComputeStackOffsets(CSymtab *symtab,
538  int param_ofs,int local_ofs)
539 {
540  assert(symtab != NULL);
541  vector<CSymbol*> slist = symtab->GetSymbols();
542 
543  // Procedure / function activation frame
544  //
545  // mem addr | container
546  // ====== caller area ======
547  // ... | ...
548  // %ebp + 16 | param 3
549  // %ebp + 12 | param 2
550  // %ebp + 8 | param 1
551  // %ebp + 4 | return addr
552  // %ebp | saved ebp
553  // ====== callee area ======
554  // %ebp - 12 | callee-saved reg
555  // %ebp - n | local var
556  // ... | ...
557 
558  size_t size = 0; // initialized by 0
559  int curParamOff = param_ofs;
560  int curLocalOff = local_ofs;
561 
562  for(int i = 0; i < slist.size(); i++) {
563  ESymbolType st = slist[i]->GetSymbolType();
564  if(st == stLocal) {
565  if (slist[i]->GetDataType() == CTypeManager::Get()->GetInt()) {
566  int padding = (4 - (size % 4)) % 4; // for alignment
567  curLocalOff -= 4 + padding; // int size
568  size += 4 + padding;
569  } else if (slist[i]->GetDataType() == CTypeManager::Get()->GetBool()) {
570  curLocalOff -= 1; // bool size
571  size += 1;
572  }
573 
574  slist[i]->SetBaseRegister("%ebp");
575  slist[i]->SetOffset(curLocalOff);
576  }
577  else if (st == stParam) {
578  slist[i]->SetBaseRegister("%ebp");
579  slist[i]->SetOffset(curParamOff);
580  curParamOff += 4;
581  }
582  }
583 
584  size += (4 - (size % 4)) % 4; // for alignment
585 
586  return size;
587 }
